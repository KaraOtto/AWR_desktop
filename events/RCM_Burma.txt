	title = burma.1.t # Government of Burma Act 1935
	desc = burma.1.d

	trigger = {
		TAG = ENG
		date > 1937.4.1

        country_exists = RAJ
		NOT = { has_global_flag = burma_act }
	}

	mean_time_to_happen = {
		days = 2
	}
	fire_only_once = yes
	immediate = {
		set_global_flag = burma_act
	}
	
	option = { #Keep playing ENG
		name = burma.1.a
		custom_effect_tooltip = burma_spawn_tooltip
		hidden_effect = {
			every_state = {
				limit = {
					is_core_of = BRM
					is_controlled_by = RAJ
				}
				transfer_state_to = BRM
			}
			ENG = { 
				set_autonomy = {
					target = BRM
					autonomous_state = autonomy_colony
					freedom_level = 0.35
				}
			 }
			 BRM = {
				set_cosmetic_tag = BRM_british_democratic
			 }
		}
	}

	option = { 
		name = rcm_burma_road.1.b
		add_political_power = -150
		FROM = {
			country_event = {
				id = EFM_independent.51 hours = 12
			}
		}
	}
}
