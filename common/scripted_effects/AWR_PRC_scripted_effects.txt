PRC_support_up = {
	if = {
		limit = {
			NOT = {
				has_idea = PRC_communist_support_low
				has_idea = PRC_communist_support_med
				has_idea = PRC_communist_support_high
			}
		}
		add_ideas = PRC_communist_support_low
	}
	else_if = {
		limit = {
			has_idea = PRC_communist_support_low
		}
		swap_ideas = {
			remove_idea = PRC_communist_support_low
			add_idea = PRC_communist_support_med
		}
	}
	else_if = {
		limit = {
			has_idea = PRC_communist_support_med
		}
		swap_ideas = {
			remove_idea = PRC_communist_support_med
			add_idea = PRC_communist_support_high
		}
	}
	else_if = {
		limit = {
			has_idea = PRC_communist_support_high
		}
		set_global_flag = PRC_communist_support_maxed_out 
	}
}
PRC_support_down = {
	if = {
		limit = {
			has_idea = PRC_communist_support_high
			has_global_flag = PRC_communist_support_maxed_out
		}
		clr_global_flag = PRC_communist_support_maxed_out
	}
	else_if = {
		limit = {
			has_idea = PRC_communist_support_high
		}
		swap_ideas = {
			remove_idea = PRC_communist_support_high
			add_idea = PRC_communist_support_med
		}
	}
	else_if = {
		limit = {
			has_idea = PRC_communist_support_med
		}
		swap_ideas = {
			remove_idea = PRC_communist_support_med
			add_idea = PRC_communist_support_low
		}
	}
	else_if = {
		limit = {
			has_idea = PRC_communist_support_low
		}
		remove_ideas = PRC_communist_support_low
	}
}

modify_parliament_seat = {
	multiply_temp_variable = { 
		temp_seats = 200 
	} #times 2 for each 0.5 unit and times 100 for percentage
	subtract_from_temp_variable = { 
		temp_member = 1 
	}

	# check to see if result overflows
	set_temp_variable = { 
		temp = parliament_seat_array^temp_member 
	}
	add_to_temp_variable = { 
		temp = temp_seats 
	}

	if = {
		limit = {
			check_variable = { 
				temp > parliament_main_array^num 
			}
		}
		subtract_from_temp_variable = { 
			temp = parliament_main_array^num 
		}
		subtract_from_temp_variable = { 
			temp_seats = temp 
		}
	}
	else_if = {
		limit = {
			check_variable = { 
				temp < 0 
			}
		}
		subtract_from_temp_variable = { 
			temp_seats = temp 
		}
	}
	
	if = {
		limit = {
			check_variable = { 
				temp_seats < 0 
			} #substract only
			OR = {
				check_variable = { 
					parliament_seat_array^0 = 200 
				} #one party is completely dominant 
				check_variable = { 
					parliament_seat_array^1 = 200 
				} #one party is completely dominant 
				check_variable = { 
					parliament_seat_array^2 = 200 
				} #one party is completely dominant 
				check_variable = { 
					parliament_seat_array^3 = 200 
				} #one party is completely dominant
			}
		}
		if = {
			limit = {
				NOT = { 
					check_variable = { 
						parliament_seat_array^0 = 200 
					} 
				} #Not the dominant one
				has_country_leader = { 
					character = PRC_zhang_guotao 
					ruling_only = yes 
				} #has 0% minority leader
			}
			add_to_temp_array = { 
				temp_array = 0 
			} #zhang reemerge
		}
		else_if = {
			limit = {
				NOT = { 
					check_variable = { 
						parliament_seat_array^1 = 200 
					} 
				} #Not the dominant one
				OR = {
					has_country_leader = { 
						character = PRC_mao_zedong ruling_only = yes 
					} #has 0% minority leader
					has_country_leader = { 
						character = PRC_zhang_wentian 
						ruling_only = yes	
					} 
					has_country_leader = { 
						character = PRC_lin_boqu 
						ruling_only = yes	
					} 
				}
			}
			add_to_temp_array = { 
				temp_array = 1 
			} #mao reemerge
		}
		else_if = {
			limit = {
				NOT = { 
					check_variable = { 
						parliament_seat_array^2 = 200 
					} 
				} #Not the dominant one
				OR = {
					has_country_leader = { 
						character = prc_bo_gu 
						ruling_only = yes 
					} 
					has_country_leader = { 
						character = prc_wang_ming 
						ruling_only = yes	
					} 
				}
			}
			add_to_temp_array = { 
				temp_array = 2 
			} #28b reemerge
		}
		else_if = {
			limit = {
				NOT = { 
					check_variable = { 
						parliament_seat_array^3 = 200 
					} 
				} #Not the dominant one
				OR = {
					has_country_leader = { 
						character = prc_peng_shuzhi 
						ruling_only = yes 
					} #has 0% minority leader
					has_country_leader = { 
						character = PRC_chen_duxiu 
						ruling_only = yes	
					} 
				}
			}
			add_to_temp_array = { 
				temp_array = 3 
			} #trot reemerge
		}
		else_if = {
			limit = {
				NOT = { 
					check_variable = { 
						temp_member = 1 
					} 
				} #is not maoist
			}
			add_to_temp_array = { 
				temp_array = 1 
			} #maoist becomes the re-emerging party
		}
		else_if = {
			limit = {
				check_variable = { 
					temp_member = 1 
				} #is maoist
			}
			add_to_temp_array = { 
				temp_array = 2 
			} #28b becomes the re-emerging party
		}
	}
	else = {
		for_loop_effect = {
			start = 0
			end = parliament_seat_array^num
			if = {
				limit = {
					check_variable = { 
						parliament_seat_array^v > 0 
					} #zero support groups don't reappear 
				}
				add_to_temp_array = { 
					temp_array = v 
				}
			}			
		}
		# remove from list of parties the party to be added/subtracted
		remove_from_temp_array = { 
			array = temp_array value = temp_member 
		}
	}
		

	#the actual code to add/subtract seats
	add_to_variable = { 
		parliament_seat_array^temp_member = temp_seats 
	}


	#make sure it doesn't go above 200(max) or below zero

	# if its add
	if = {
		limit = {
			check_variable = { 
				temp_seats > 0 
			}
		}
		# loop until seats to be added is 0
		while_loop_effect = {
			limit = {
				check_variable = { 
					temp_seats > 0 
				}
			}

			# loops through remaining parties, subtracting 1 seat each time
			for_loop_effect = {
				start = 0
				end = temp_array^num

				if = {
					limit = {
						check_variable = { 
							parliament_seat_array^temp_array^v > 0 
						} # party must have more than 0 seats to be subtracted
						check_variable = { 
							temp_seats > 0 
						} # seats to be added is bigger than 0 currently
					}
					subtract_from_variable = { 
						parliament_seat_array^temp_array^v = 1 
					} # subtract from party 1 seat
					subtract_from_temp_variable = { 
						temp_seats = 1 
					} # subtract from seats to be added
				}
			}
		}
		
	}
	# if its subtract
	else_if = {
		limit = {
			check_variable = { 
				temp_seats < 0 
			}
			#check_variable = { parliament_seat_array^temp_member < 0 }
		}
		# loop until seats to be subtracted is 0
		while_loop_effect = {
			limit = {
				check_variable = { 
					temp_seats < 0 
				}
			}
			# loops through remaining parties, adding 1 seat each time
			for_loop_effect = {
				start = 0
				end = temp_array^num
				if = {
					limit = {
						check_variable = { 
							parliament_seat_array^temp_array^v < parliament_main_array^num 
						} # party must have less than max seats to be added
						check_variable = { 
							temp_seats < 0 
						} # seats to be subtracted is less than 0 currently
					}
					add_to_variable = { 
						parliament_seat_array^temp_array^v = 1 
					} # add to party 1 seat
					add_to_temp_variable = { 
						temp_seats = 1 
					} # add to seats to be subtracted
				}
			}
		}
	}
	# do nothing if 0
	# give party new seat value
	
	
	# Modify what shows up in the gui
	reload_parliament_gui = yes
}

X_xchinese_pick_propaganda_campaign_slot_effect = {
	hidden_effect = {
		if = {
			limit = {
				check_variable = { 
					X_xchinese_poster_left_icon_frame = X_xchinese_propaganda_empty_slot_index 
				}
			}
			set_variable = { 
				X_xchinese_poster_left_icon_frame = X_xchinese_new_propaganda_campaign 
			}
			#Ageing posters
			set_variable = { 
				X_xchinese_propaganda_left_poster_to_age = X_xchinese_poster_left_icon_frame 
			}
			hidden_effect = {
				country_event = { 
					id = xpand_xchinese_capmaign_events.101 
					days = X_xchinese_default_propaganda_campaign_old_days 
				}
				
				if = { 
					limit = {
						has_country_flag = X_xchinese_debug_fast_propaganda_campaigns_flag
					}
					set_country_flag = { 
						flag = X_xchinese_propaganda_left_days_remaining_flag 
						days = 5 
					}
				}
				else = { 
					set_country_flag = { 
						flag = X_xchinese_propaganda_left_days_remaining_flag 
						days = 180 
					}
				}
			}
		}
		else_if = {
			limit = {
				check_variable = { 
					X_xchinese_poster_center_icon_frame = X_xchinese_propaganda_empty_slot_index 
				}
			}
			set_variable = { 
				X_xchinese_poster_center_icon_frame = X_xchinese_new_propaganda_campaign 
			}
			#Ageing posters
			set_variable = { 
				X_xchinese_propaganda_center_poster_to_age = X_xchinese_poster_center_icon_frame 
			} 
			hidden_effect = {
				country_event = { 
					id = xpand_xchinese_capmaign_events.102 
					days = X_xchinese_default_propaganda_campaign_old_days 
				}
				
				if = { 
					limit = {
						has_country_flag = X_xchinese_debug_fast_propaganda_campaigns_flag
					}
					set_country_flag = { 
						flag = X_xchinese_propaganda_center_days_remaining_flag 
						days = 5 
					}
				}
				else = { 
					set_country_flag = { 
						flag = X_xchinese_propaganda_center_days_remaining_flag 
						days = 180 
					} 
				}
			}
		}
		else_if = {
			limit = {
				check_variable = { X_xchinese_poster_right_icon_frame = X_xchinese_propaganda_empty_slot_index }
			}
			set_variable = { X_xchinese_poster_right_icon_frame = X_xchinese_new_propaganda_campaign }
			#Ageing posters
			set_variable = { X_xchinese_propaganda_right_poster_to_age = X_xchinese_poster_right_icon_frame } #This is used to prevent the system to break if a propaganda campaign is cancelled
			hidden_effect = {
				country_event = { 
					id = xpand_xchinese_capmaign_events.103 
					days = X_xchinese_default_propaganda_campaign_old_days 
				}
				
				if = { #Debug -> Faster time
					limit = {
						has_country_flag = X_xchinese_debug_fast_propaganda_campaigns_flag
					}
					set_country_flag = { 
						flag = X_xchinese_propaganda_right_days_remaining_flag 
						days = 5 
					}
				}
				else = { #Regular time
					set_country_flag = { 
						flag = X_xchinese_propaganda_right_days_remaining_flag days = 180 
					} 
					
				}
			}
		}
		set_variable = { 
			X_xchinese_new_propaganda_campaign = 0 
		} # Reset var
		add_to_variable = { 
			X_xchinese_propaganda_campaigns_taken = 1 
		} 
	}
}
