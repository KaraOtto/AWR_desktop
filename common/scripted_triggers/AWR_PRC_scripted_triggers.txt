leader_is_zhang_guotao = {	
	
	OR = {
		has_country_leader = {
			character = PRC_zhang_guotao 
			ruling_only = yes
		}
		has_country_leader = {#guo
			character = PRC_quanguorenda 
			ruling_only = yes
		}
	}
}
