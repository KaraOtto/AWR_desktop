on_actions = {  #整风运动
	on_daily = {
		effect = {
            if = {
                limit = {
                    check_variable = {
                        variable = yan_an_zheng_feng_prog_var
                        value > 100
                    }
                }
                set_variable = {
                    variable = yan_an_zheng_feng_prog_var
                    value = 100
                }
            }
            if = {
                limit = {
                    check_variable = {
                        variable = yan_an_zheng_feng_prog_var
                        value < 0
                    }
                }
                set_variable = {
                    variable = yan_an_zheng_feng_prog_var
                    value = 0
                }
            }
        }
    }
}

on_actions = {  #解放战争
    on_war_relation_added = {   #抢占主动权
		effect = {
			if = {
				limit = {
					ROOT = { 
                        tag = PRC 
                    }
					FROM = { 
                        tag = CHI 
                    }
					NOT = { 
                        has_global_flag = the_Chinese_civil_war
                    } 
                    date > 1938.1.1
				}
				set_global_flag = the_Chinese_civil_war
				complete_national_focus = PRC_seize_the_initiative
			}
		}
	}

    on_war_relation_added = {   #战略防御
		effect = {
			if = {
				limit = {
					ROOT = {

                    }
