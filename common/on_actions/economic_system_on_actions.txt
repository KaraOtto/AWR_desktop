on_actions = {
	on_monthly = {	#月度变化
		effect = {
			every_country = {
				set_temp_variable = {	#临时变量
					var = economy_development_monthly
					value = modifier@AWR_economy_development_monthly
				}

				add_to_variable = {
					var = economy_development_index
					value = economy_development_monthly
				}				
			}
		}
	}

	on_daily = {	#经济发展提升
		effect = {
			every_country = {
				if = {
					limit = {
						check_variable = { 
							var = economy_development_index
							value = 1
							compare = greater_than_or_equals
						}
					}

					add_to_variable = {
						var = economy_development_index
						value = -1.00
					}

					economy_situation_up = yes
				}
			}
		}
	}

	on_daily = {	#经济发展恶化
		effect = {
			every_country = {
				if = {
					limit = {
						check_variable = { 
							var = economy_development_index
							value = 0
							compare = less_than
						}
					}

					add_to_variable = {
						var = economy_development_index
						value = 1.00
					}

					economy_situation_down = yes

				}
			}
		}
	}
}
