on_actions = {
	on_startup = {
		effect = {
			every_country = {
				if = {
					limit = {
						is_ai = no
					}
					country_event = { 
						id = start_events.1
					}
				}
			}
		}
	}

	on_startup = {
		effect = {
			every_country = {
				if = {
					limit = {
						has_government = communism
					}
					add_ideas = materialism
					else_if = {
						limit = {
							OR = {
								has_government = democratic
								tag = CHI
							}
						}
						add_ideas = idealism

						else = {
							add_ideas = both_materfialism_and_idealism
						}
					}
				}

				if = {
					limit = {
						OR = {
							has_government = fascism
							tag = CHI
						}
					}
					add_ideas = unfriendly

					else = {
						add_ideas = neither_friendly_nor_unfriendly
					}
					
				}

				if = {
					limit = {
						has_government = fascism
					}
					
					add_ideas = expansionism
					
					else_if = {
						limit = { 
							has_government = democratic
						}
						add_ideas = pacifism

						else = {
							add_ideas = between_pacifism_and_expansionism
						}
					}
				}

			}
		}
	}
}
